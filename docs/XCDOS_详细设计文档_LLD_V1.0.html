LLD V1.0

基于 TDD V1.0（高阶设计）展开的前端组件级 + 后端函数级的详细设计。覆盖组件树、状态管理、UseCase 编排、定时任务与缓存键设计，面向单模块编码。

前端：React + Next.js + TypeScript 状态：Zustand + React Query 后端：NestJS + DDD 分层 缓存：Redis 多级策略

一、设计范围

本文档为 TDD V1.0 的下沉补充——TDD 定义了模块边界和职责（What），LLD 定义组件/函数级实现（How）。

后端

五、核心 UseCase 编排

5.1 CreateGoalUseCase 执行步骤

async execute(dto: CreateGoalDto, userId: string): Promise<Goal> {
  // 1. 校验负责人存在且状态 active
  const owner = await this.userRepo.findById(dto.ownerId);
  if (!owner || owner.status !== 'active') throw new NotFoundException('负责人不存在或已禁用');

  // 2. 校验组织存在
  const org = await this.orgRepo.findById(dto.orgId);

  // 3. 如果有父目标，校验父目标存在且为 active
  if (dto.parentId) {
    const parent = await this.goalRepo.findById(dto.parentId);
    if (!parent || parent.status === 'cancelled') throw new BadRequestException('父目标不可用');
  }

  // 4. 创建 Goal 实体
  const goal = Goal.create({ ...dto, createdBy: userId, status: 'draft' });

  // 5. 持久化
  const saved = await this.goalRepo.save(goal);

  // 6. 发布领域事件
  await this.eventBus.publish(new GoalCreatedEvent(saved));

  // 7. 写审计日志
  await this.auditService.log({ actorId: userId, action: 'CREATE', resourceType: 'goal', resourceId: saved.id });

  return saved;
}

5.2 SubmitFeedbackUseCase 执行步骤

async execute(dto: SubmitFeedbackDto, userId: string): Promise<Feedback> {
  // 1. 校验任务存在且用户为 owner 或协作者
  const task = await this.taskRepo.findById(dto.taskId);
  if (task.ownerId !== userId) throw new ForbiddenException('仅任务负责人可提交反馈');

  // 2. 检查今日是否已有反馈
  const existing = await this.feedbackRepo.findTodayByTask(dto.taskId, userId);
  if (existing) {
    // 更新已有反馈（非重复创建）
    existing.update(dto);
    return this.feedbackRepo.save(existing);
  }

  // 3. 创建反馈
  const feedback = Feedback.create({ ...dto, userId, submittedAt: new Date() });
  const saved = await this.feedbackRepo.save(feedback);

  // 4. 触发异步质量评估
  await this.agentRunService.startRun('feedback_quality', 'event', { feedbackId: saved.id });

  // 5. 发布 FeedbackSubmitted 事件（驾驶舱 KPI 刷新）
  await this.eventBus.publish(new FeedbackSubmittedEvent(saved));

  return saved;
}

5.3 GenerateDecisionReportUseCase 执行步骤

async execute(caseId: string): Promise<DecisionReport> {
  const case = await this.caseRepo.findByIdWithRelations(caseId);
  // case = { problem, hypotheses[], evaluation, roiSimulation, plan }

  // 1. 校验全流程阶段完成
  if (case.stage !== 'calculate') throw new BadRequestException('请先完成评估与试算');

  // 2. 组装报告
  const report: DecisionReport = {
    problem: { title: case.problem.title, severity: case.problem.severity },
    hypotheses: case.hypotheses.map(h => ({ content: h.content, confidence: h.confidence, counter: h.counterExample })),
    evaluation: { resource: case.evaluation.resourceScore, time: case.evaluation.timeScore, risk: case.evaluation.riskScore, feasibility: case.evaluation.feasibilityScore },
    roi: { cost: case.roiSimulation.cost, revenue: case.roiSimulation.revenue, roi: case.roiSimulation.roi, paybackDays: case.roiSimulation.paybackDays },
    plan: case.plan ? { title: case.plan.title, description: case.plan.description } : null,
    generatedAt: new Date(),
  };

  // 3. 更新案例状态
  case.stage = 'report';
  case.status = 'resolved';
  await this.caseRepo.save(case);

  return report;
}

缓存防护实现要点

// 缓存穿透：布隆过滤器 + 空值缓存
const exists = await bloomFilter.mayContain(`goal:${id}`);
if (!exists) throw new NotFoundException('目标不存在');
// 如果 DB 返回 null，缓存空标记 60s

// 缓存击穿：热点 Key 互斥锁

if (!locked) { await sleep(100); return getFromCache(key); } // 自旋等锁释放
